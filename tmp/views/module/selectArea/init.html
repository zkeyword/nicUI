接口说明：

地区初始化的接口是initValue，initValue的值来至于服务器传过来的值，必须跟selectArea_data.js的值是一致的。

应用代码：

/*html 代码*/
<div id="lt-wrap" class="fn-clear">
    <span class="fn-left">
        <select class="ui-select ui-select-130" id="province"></select>
    </span>
    <span class="fn-left">
        <select class="ui-select ui-select-130" id="city"></select>
    </span>
    <span class="fn-left">
        <select class="ui-select ui-select-130" id="county"></select>
    </span>
</div>

/*js 地区数据*/
<script src="/resources/nic/build/js/app/selectArea_data.js"></script>

/*js 代码*/
nic.ui.selectArea({
	data: areaData,
	initValue: {
		province:'120000',
		city:'120100',
		county:'120101'
	}
	province: '#province',
	city: '#city',
	county: '#county'
})
