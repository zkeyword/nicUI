接口说明：

target为check类型的选择器，使用时设置为jquery的选择器即可，同时请在html设置好checkbox、radio的name值。

应用代码：

/*html 代码*/
<div class="ui-form fn-clear">
    <label class="ui-pl10">
        <input type="checkbox" class="ui-checkbox" name="testCheckbox[]" value="多选选项1" />
        <i class="ui-label ui-checkbox-label">多选选项1</i>
    </label>
    <label class="ui-pl10">
        <input type="checkbox" class="ui-checkbox " name="testCheckbox[]" value="多选选项2"  />
        <i class="ui-label ui-checkbox-label">多选选项2</i>
    </label>
    <label class="ui-pl10">
        <input type="radio" class="ui-radio" name="testRaio" value="单选选项3" />
        <i class="ui-label ui-radio-label">单选选项3</i>
    </label>
    <label class="ui-pl10">
        <input type="radio" class="ui-radio " name="testRaio" value="单选选项4"  />
        <i class="ui-label ui-radio-label">单选选项4</i>
    </label>
</div>

/*js 代码*/
var checkbox = nic.ui.check({
	target:':radio'
});

var checkbox = nic.ui.check({
	target:':checkbox'
})
