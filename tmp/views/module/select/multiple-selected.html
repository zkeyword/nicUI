接口说明：

多选设置已选的接口是selectedData，要求的数据格式json，其格式与数据源的格式是一致的。

应用代码：

/*html 代码*/
<div class="fn-clear">
	<input type="hidden" name="multiple" id="multiple" />
	<a href="#" class="ui-btn ui-btnMain" id="refresh">点击刷新</a>
</div>

/*js 代码*/
var select = nic.ui.select2({
	target:'#multiple',
	type:'multiple',
	//data:data, 静态数据
	ajax:{
		url: 'http://127.0.0.1/nic/doc/tmp/treeData.php',
		data:{
			index:'1'
		}
	}
});
