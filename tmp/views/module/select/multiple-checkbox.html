接口说明：

多选复选框的API是checkbox，默认为false。

应用代码：

/*html 代码*/
<div class="fn-clear">
	<input type="hidden" name="multiple" id="multiple" />
</div>

/*js 代码*/
var select = nic.ui.select2({
	target:'#multiple',
	type:'multiple',
	//data:data, 静态数据
	checkbox:true,
	ajax:{
		url: 'http://127.0.0.1/nic/doc/tmp/select.php',
		data:{
			index:'1'
		}
	}
});
