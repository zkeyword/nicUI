接口说明：

下拉树时直接将接口type设置tree即可，其中selectedData是设置初始化选中值的ID，传入时是数组格式，该数组值的数据类型必须与数据是一直的。

该功能依赖tree控件，单独使用时请引用tree.js。

应用代码：

/*html 代码*/
<div class="fn-clear">
<input type="hidden" name="tree" id="tree"/>
</div>

/*js 代码*/
var tree = nic.ui.tree({
	//data:data,
	target:$('#tree'),
	ajax:{
		url: 'http://127.0.0.1/nic/doc/tmp/treeData.php',
		data:{
			index:'1'
		}
	},
	selected:[13, 11],
	onClick:function(obj, data){
		console.log(obj, data)
	},
	//onRightClick:function(){
	//	alert(11)
	//},
	onLoad:function(){
		//alert('1');
	},
	onMouseOver:function(){
		
	},
	onMouseOut:function(){
	
	}
});
