接口说明：

清空重置的API是reset，该接口需要等selectArea对象生成之后才能使用。

应用代码：

/*html 代码*/
<div id="lt-wrap" class="fn-clear">
    <span class="fn-left">
        <select class="ui-select ui-select-130" id="province"></select>
    </span>
    <span class="fn-left">
        <select class="ui-select ui-select-130" id="city"></select>
    </span>
    <span class="fn-left">
        <select class="ui-select ui-select-130" id="county"></select>
    </span>
	<a href="#" id="reset" class="ui-btn ui-btnMain">清空重置</a>
</div>

/*js 地区数据*/
<script src="/resources/nic/build/js/app/selectArea_data.js"></script>

/*js 代码*/
var area = nic.ui.selectArea({
	data: areaData,
	initValue: {
		province:'120000',
		city:'120100',
		county:'120101'
	},
	province: '#province',
	city: '#city',
	county: '#county'
});

$('#reset').click(function(){
	area.reset();
	return false;
});
