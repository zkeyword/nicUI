接口说明：

刷新多选选项的接口是refresh，需在多选对象生成之后才能使用。

应用代码：

/*html 代码*/
<div class="fn-clear">
	<input type="hidden" name="multiple" id="multiple" />
	<a href="#" class="ui-btn ui-btnMain" id="refresh">点击刷新</a>
</div>

/*js 代码*/
var select = nic.ui.select2({
	target:'#multiple',
	type:'multiple',
	//data:data, 静态数据
	ajax:{
		url: 'http://127.0.0.1/nic/doc/tmp/treeData.php',
		data:{
			index:'1'
		}
	}
});
$('#refresh').click(function(){
	select.refresh({
		ajax:{
			url: 'http://127.0.0.1/nic/doc/tmp/select.php',
			data:{
				index:'2'
			}
		}
	});
	return false;
});

调用效果：

点击刷新
